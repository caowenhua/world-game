### 1.4 瓦片渲染规范

```typescript
// 草地瓦片
drawGrassTile(ctx, sx, sy, tileX, tileY, frame) {
  // 主色填充
  ctx.fillStyle = '#7ECF7B';
  ctx.fillRect(sx, sy, 32, 32);
  
  // 随机草叶（基于tileX/tileY种子）
  ctx.fillStyle = '#5FB55F';
  for (let i = 0; i < 4; i++) {
    const gx = sx + ((seed + i * 3) % 28) + 2;
    const gy = sy + ((seed + i * 7) % 28) + 2;
    const sway = Math.sin(frame * 0.05 + i + tileX * 0.3) * 1;
    ctx.fillRect(gx + sway, gy, 2, 4);
  }
}
```
